nome:
    A Logical Labyrinth
sentenças:
    count(m[1..9]) = 1
    for x in [1..9]: m[x] -> p[x]
    for x in [1..9]: t[x] -> !p[x]
    !n[8]

    p[1] <=> m[1] | m[3] | m[5] | m[7] | m[9]
    p[2] <=> n[2]
    p[3] <=> p[5] | !p[8]
    p[4] <=> !p[1]
    p[5] <=> p[2] | p[4]
    p[6] <=> !p[3]
    p[7] <=> !m[1]
    p[8] <=> t[8] & n[9]
    p[9] <=> t[9] & !p[6]

solução:
    "A moça está atrás da porta " + [for x in [1..9]: x if m[x]]
